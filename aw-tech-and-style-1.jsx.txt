/* ---------------- Fragrance "AI" generation engine ----------------
   This simulates an AI product generator (image + copy + SEO) using
   deterministic, template-driven synthesis so the experience is fully

const NOTE_LIBRARY = {
  top: ["Calabrian Bergamot", "Pink Pepper", "Sicilian Mandarin", "Cardamom", "White Currant", "Bulgarian Rose Petal", "Saffron Thread", "Green Fig Leaf"],
  heart: ["Damask Rose", "Iris Pallida", "Jasmine Sambac", "Tuberose", "Orris Root", "Violet Leaf", "Osmanthus", "Black Tea Accord"],
  base: ["Mysore Sandalwood", "Madagascar Vanilla", "Amber Resin", "Oud Wood", "Tonka Bean", "Vetiver", "White Musk", "Patchouli"],
};

const MOODS = ["Seductive", "Radiant", "Mysterious", "Serene", "Bold", "Romantic", "Magnetic", "Ethereal"];
const OCCASIONS = ["Evening Gala", "Daily Signature", "Date Night", "Office Elegance", "Wedding", "Summer Escape"];
const GENDERS = ["Femme", "Homme", "Unisex"];

function pick(arr, n, seedOffset = 0) {
  const copy = [...arr];
  const out = [];
  for (let i = 0; i < n; i++) {
    const idx = (i * 7 + seedOffset * 3 + copy.length) % copy.length;
    out.push(copy.splice(idx % copy.length, 1)[0]);
  }
  return out;
}

function generateFragranceProfile(name, seed = 0) {
  const top = pick(NOTE_LIBRARY.top, 2, seed);
  const heart = pick(NOTE_LIBRARY.heart, 2, seed + 1);
  const base = pick(NOTE_LIBRARY.base, 2, seed + 2);
  const mood = MOODS[seed % MOODS.length];
  const occasion = OCCASIONS[(seed + 2) % OCCASIONS.length];
  const gender = GENDERS[seed % GENDERS.length];
  return { top, heart, base, mood, occasion, gender };
}

// Simulated "AI description writer"

  const { top, heart, base, mood } = profile;
  const templates = [

Beneath it all, ${base[0]} and ${base[1]} linger like a memory that refuses to fade. ${mood} in character, this is a fragrance for those who prefer to be remembered, not introduced.`,
    `There is a quiet audacity to ${name}. ${top[0]} sparks the opening, ${heart[0]} and ${heart[1]} carry the soul of it, and a base built on ${base[0]} and ${base[1]} ensures it never truly leaves the skin. ${mood} and unapologetic — this is scent as signature.`,
    `${name} begins with ${top[0]} and ${top[1]}, sharp and alive, then folds into ${heart[0]} at the heart. The drydown — ${base[0]} laced with ${base[1]} — is where the story lingers longest. A ${mood.toLowerCase()} composition, made for skin that intends to be noticed.`,
  ];
  return templates[(name.length + top.length) % templates.length];
}

// Simulated "AI SEO generator"
function generateSEO(name, profile) {
  const { mood, occasion, gender } = profile;
  return {
    title: `${name} — Luxury ${gender} Eau de Parfum | A_W Perfume`,
    metaDescription: `Discover ${name}, a ${mood.toLowerCase()} fragrance crafted for ${occasion.toLowerCase()}. Notes of ${profile.top[0]} and ${profile.base[0]}. Free worldwide shipping.`,
    keywords: [name.toLowerCase(), `luxury ${gender.toLowerCase()} perfume`, `${mood.toLowerCase()} fragrance`, `${occasion.toLowerCase()} perfume`, "designer eau de parfum", "A_W Perfume"],
  };
}

/* ---------------- REAL AI description + SEO writer ----------------
   This calls the live Anthropic API (Claude) to generate genuinely
   unique product copy and SEO metadata on demand. Falls back to the
   template generator above if the call fails (offline, rate-limited,
   etc.) so the UI never breaks.
------------------------------------------------------------------- */

async function generateDescriptionAndSEO_AI(name, profile) {
  const prompt = `You are the senior copywriter for A_W Perfume, an ultra-luxury fragrance house (think Dior/Chanel tier). Write product copy for a perfume called "${name}".

Fragrance profile to use as creative material (you may select which notes to foreground, but stay consistent with these):
- Top notes: ${profile.top.join(", ")}
- Heart notes: ${profile.heart.join(", ")}
- Base notes: ${profile.base.join(", ")}
- Mood: ${profile.mood}
- Best occasion: ${profile.occasion}
- Gender category: ${profile.gender}

Write:
1. A product description (3-5 sentences, evocative, editorial, luxury fashion-house tone — no clichés like "unleash" or "elevate", no exclamation marks)
2. An SEO title tag (under 60 characters, include the brand name "A_W Perfume")
3. An SEO meta description (under 155 characters)
4. 5 SEO keywords as an array of short phrases

Respond ONLY with raw JSON, no markdown fences, no preamble, in exactly this shape:
{"description": "...", "seoTitle": "...", "seoMetaDescription": "...", "seoKeywords": ["...", "...", "...", "...", "..."]}`;

  const response = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-6",
      max_tokens: 1000,
      messages: [{ role: "user", content: prompt }],
    }),
  });

  if (!response.ok) throw new Error(`AI request failed with status ${response.status}`);

  const data = await response.json();
  const text = data.content
    .map((block) => (block.type === "text" ? block.text : ""))
    .filter(Boolean)
    .join("\n");

  const cleaned = text.replace(/```json|```/g, "").trim();
  const parsed = JSON.parse(cleaned);

  return {
    description: parsed.description,
    seo: {
      title: parsed.seoTitle,
      metaDescription: parsed.seoMetaDescription,
      keywords: parsed.seoKeywords,
    },
  };
}

// Simulated "AI image generator" — produces a deterministic, unique gradient
// bottle visual per product so every item looks distinct without real image-gen.
function bottleArt(seed, accent) {
  const hues = [
    ["#1a1612", "#3d2f1f", "#d4af6a"],
    ["#0e0d0c", "#2a2118", "#c89a5b"],
    ["#16110d", "#4a3826", "#e8c688"],
    ["#120f0c", "#352618", "#caa86a"],
  ];
  return hues[seed % hues.length];
}

/* ---------------- Demo-only password hashing ----------------
   IMPORTANT: this is NOT real cryptographic security. It's a simple
   string transform so plain-text passwords aren't sitting in state
   completely as-typed. Real apps must hash + verify passwords on a
   real backend server (e.g. bcrypt/argon2 via Node, or a managed
   auth provider like Supabase/Firebase Auth/Clerk). Never trust
   client-side-only auth for a real business — see README at the
   bottom of this file.
------------------------------------------------------------------- */
function simpleHash(str) {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = (hash << 5) - hash + str.charCodeAt(i);
    hash |= 0;
  }
  return `h${hash}`;
}

const StoreContext = createContext(null);
function useStore() {
  return useContext(StoreContext);
}

const TAX_RATE = 0.08;
const SHIPPING_FLAT = 18;
const FREE_SHIP_THRESHOLD = 250;

function StoreProvider({ children }) {
  const [catalog, setCatalog] = useState(() => buildCatalog());
  const [cart, setCart] = useState([]); // {id, qty}
  const [orders, setOrders] = useState([]);
  const [wishlist, setWishlist] = useState([]);
  const [viewHistory, setViewHistory] = useState([]); // for "smart recommendations"
  const [discountCode, setDiscountCode] = useState("");
  const [aiGenerating, setAiGenerating] = useState({}); // { [productId]: boolean }
  const [aiError, setAiError] = useState({}); // { [productId]: string | null }
  const orderCounter = useRef(1000);

  /* -------- Auth (demo-only — see README at bottom for production notes) -------- */
  // Customers: stored in memory as { name, email, passwordHash }.
  // Admin: single hardcoded demo account, password set below.
  const [customers, setCustomers] = useState([]); // registered accounts (in-memory)
  const [currentUser, setCurrentUser] = useState(null); // logged-in customer
  const [isAdmin, setIsAdmin] = useState(false); // admin session flag
  const [authError, setAuthError] = useState("");
  const ADMIN_EMAIL = "admin@awperfume.com";
  const ADMIN_PASSWORD_HASH = simpleHash("AWAdmin#2026"); // demo admin password

  function signup({ name, email, password }) {
    setAuthError("");
    if (!name || !email || !password) { setAuthError("All fields are required."); return false; }
    if (password.length < 6) { setAuthError("Password must be at least 6 characters."); return false; }
    if (customers.some((c) => c.email.toLowerCase() === email.toLowerCase())) {
      setAuthError("An account with this email already exists."); return false;
    }
    const newCustomer = { name, email, passwordHash: simpleHash(password) };
    setCustomers((c) => [...c, newCustomer]);
    setCurrentUser({ name, email });
    return true;
  }

  function login({ email, password }) {
    setAuthError("");
    const found = customers.find((c) => c.email.toLowerCase() === email.toLowerCase());
    if (!found || found.passwordHash !== simpleHash(password)) {
      setAuthError("Invalid email or password.");
      return false;
    }
    setCurrentUser({ name: found.name, email: found.email });
    return true;
  }

  function logout() {
    setCurrentUser(null);
  }

  function adminLogin({ email, password }) {
    setAuthError("");
    if (email.toLowerCase() !== ADMIN_EMAIL || simpleHash(password) !== ADMIN_PASSWORD_HASH) {
      setAuthError("Invalid admin credentials.");
      return false;
    }
    setIsAdmin(true);
    return true;
  }

  function adminLogout() {
    setIsAdmin(false);
  }

  /* -------- Store customization (theme) -------- */
  const [theme, setTheme] = useState({
    accent: "#d4af6a",
    storeName: "A_W",
    tagline: "TECH & STYLE",
  });

  /* -------- Reviews system -------- */
  const [reviews, setReviews] = useState({}); // { [productId]: [{name, rating, comment, date}] }

  function addReview(productId, review) {
    setReviews((r) => ({
      ...r,
      [productId]: [{ ...review, date: new Date().toISOString() }, ...(r[productId] || [])],
    }));
  }

  /* -------- Blog posts -------- */
  const [blogPosts] = useState([
    {
      id: "blog-1",
      title: "Introducing Noir Élixir — Our Most Requested Scent Returns",
      date: "2026-06-28",
      tag: "New Launch",
      excerpt: "After months of anticipation, Noir Élixir is back in a limited 100ml edition. Here's everything you need to know about the relaunch, what changed, and how to get yours before it sells out.",
      readTime: "3 min read",
    },
    {
      id: "blog-2",
      title: "A_W Tech & Style Expands to 40+ Countries — Worldwide Shipping Now Live",
      date: "2026-06-15",
      tag: "Announcement",
      excerpt: "We're thrilled to announce that A_W Tech & Style now ships to over 40 countries worldwide with free delivery on orders above $250. Our luxury fragrances are now accessible to customers across Europe, Asia, and the Americas.",
      readTime: "2 min read",
    },
    {
      id: "blog-3",
      title: "Summer 2026 Drop: Three New Compositions Inspired by the Mediterranean",
      date: "2026-06-01",
      tag: "New Launch",
      excerpt: "This summer we're releasing three limited-edition fragrances drawing from the sun-soaked coasts of Amalfi, Ibiza, and Santorini. Each bottle tells a story — and each story starts with a single spray.",
      readTime: "4 min read",
    },
    {
      id: "blog-4",
      title: "Behind the Bottle: How Our AI Studio Creates Fragrance Profiles",
      date: "2026-05-20",
      tag: "Announcement",
      excerpt: "For the first time, we're sharing a look inside our AI-powered product studio — how we go from a fragrance concept to a complete product listing with notes, mood, occasion, and SEO copy in under 60 seconds.",
      readTime: "5 min read",
    },
  ]);

  function recordView(id) {
    setViewHistory((h) => [id, ...h.filter((x) => x !== id)].slice(0, 10));
  }

  function addToCart(id, qty = 1) {
    setCart((c) => {
      const existing = c.find((i) => i.id === id);
      if (existing) return c.map((i) => (i.id === id ? { ...i, qty: i.qty + qty } : i));
      return [...c, { id, qty }];
    });
  }
  function updateQty(id, qty) {
    setCart((c) => (qty <= 0 ? c.filter((i) => i.id !== id) : c.map((i) => (i.id === id ? { ...i, qty } : i))));
  }
  function removeFromCart(id) {
    setCart((c) => c.filter((i) => i.id !== id));
  }
  function toggleWishlist(id) {
    setWishlist((w) => (w.includes(id) ? w.filter((x) => x !== id) : [...w, id]));
  }

  function cartLines() {
    return cart.map((i) => ({ ...i, product: catalog.find((p) => p.id === i.id) })).filter((l) => l.product);
  }

  function pricing(discountPct = 0) {
    const lines = cartLines();
    const subtotal = lines.reduce((s, l) => s + l.product.price * l.qty, 0);
    const discount = subtotal * (discountPct / 100);
    const afterDiscount = subtotal - discount;
    const shipping = afterDiscount >= FREE_SHIP_THRESHOLD || afterDiscount === 0 ? 0 : SHIPPING_FLAT;
    const tax = afterDiscount * TAX_RATE;
    const total = afterDiscount + shipping + tax;
    return { subtotal, discount, shipping, tax, total };
  }

  function placeOrder({ customer, payment, discountPct }) {
    const lines = cartLines();
    if (lines.length === 0) return null;
    const totals = pricing(discountPct);
    const id = `AW-${orderCounter.current++}`;
    const order = {
      id,
      date: new Date().toISOString(),
      customer,
      payment,
      items: lines.map((l) => ({ id: l.id, name: l.product.name, qty: l.qty, price: l.product.price, sizeMl: l.product.sizeMl })),
      totals,
      status: "Processing",
      history: [{ status: "Processing", at: new Date().toISOString() }],
    };
    setOrders((o) => [order, ...o]);
    // decrement stock
    setCatalog((cat) => cat.map((p) => {
      const line = lines.find((l) => l.id === p.id);
      return line ? { ...p, stock: Math.max(0, p.stock - line.qty) } : p;
    }));
    setCart([]);
    return order;
  }

  function advanceOrderStatus(orderId, status) {
    setOrders((os) => os.map((o) => o.id === orderId ? {
      ...o, status, history: [...o.history, { status, at: new Date().toISOString() }]
    } : o));
  }

  function updateStock(productId, stock) {
    setCatalog((cat) => cat.map((p) => p.id === productId ? { ...p, stock } : p));
  }

  // Real AI-powered regeneration: calls Claude to write fresh description + SEO.
  // Falls back to the local template generator if the API call fails.
  async function regenerateProduct(productId) {
    setAiError((e) => ({ ...e, [productId]: null }));
    setAiGenerating((g) => ({ ...g, [productId]: true }));

    const current = catalog.find((p) => p.id === productId);
    if (!current) {
      setAiGenerating((g) => ({ ...g, [productId]: false }));
      return;
    }

    // Shuffle the fragrance profile each regeneration for variety, same as before.
    const seed = Math.floor(Math.random() * 1000);
    const profile = generateFragranceProfile(current.name, seed);
    const art = bottleArt(seed, "#d4af6a");

    try {
      const { description, seo } = await generateDescriptionAndSEO_AI(current.name, profile);
      setCatalog((cat) => cat.map((p) => p.id === productId ? { ...p, profile, description, seo, art } : p));
    } catch (err) {
      // Fallback to local generator so the UI never breaks if the AI call fails.
      const description = generateDescription(current.name, profile);
      const seo = generateSEO(current.name, profile);
      setCatalog((cat) => cat.map((p) => p.id === productId ? { ...p, profile, description, seo, art } : p));
      setAiError((e) => ({ ...e, [productId]: "AI generation failed — used local fallback copy instead." }));
    } finally {
      setAiGenerating((g) => ({ ...g, [productId]: false }));
    }
  }

  const [newProductGenerating, setNewProductGenerating] = useState(false);
  const [newProductError, setNewProductError] = useState(null);

  // Creates a brand-new fragrance using the real AI writer end-to-end:
  // given just a name (and optional price/size), Claude generates the
  // fragrance profile's description + SEO; notes/mood/occasion are
  // assigned locally for catalog consistency, then woven into the prompt.
  async function addProduct({ name, price, sizeMl }) {
    setNewProductError(null);
    setNewProductGenerating(true);
    const seed = catalog.length + Math.floor(Math.random() * 50);
    const profile = generateFragranceProfile(name, seed);
    const art = bottleArt(seed, "#d4af6a");
    const id = `aw-${Date.now()}`;

    try {
      const { description, seo } = await generateDescriptionAndSEO_AI(name, profile);
      setCatalog((cat) => [
        ...cat,
        {
          id, name, price: Number(price) || 250, sizeMl: Number(sizeMl) || 50,
          profile, description, seo, art, stock: 25, rating: "4.5", reviews: 0, featured: false,
        },
      ]);
    } catch (err) {
      const description = generateDescription(name, profile);
      const seo = generateSEO(name, profile);
      setCatalog((cat) => [
        ...cat,
        {
          id, name, price: Number(price) || 250, sizeMl: Number(sizeMl) || 50,
          profile, description, seo, art, stock: 25, rating: "4.5", reviews: 0, featured: false,
        },
      ]);
      setNewProductError("AI generation failed — used local fallback copy instead.");
    } finally {
      setNewProductGenerating(false);
    }
  }

  const value = {
    catalog, cart, orders, wishlist, viewHistory, discountCode, setDiscountCode,
    addToCart, updateQty, removeFromCart, toggleWishlist, cartLines, pricing,
    placeOrder, advanceOrderStatus, updateStock, regenerateProduct, recordView,
    aiGenerating, aiError, addProduct, newProductGenerating, newProductError,
    currentUser, isAdmin, authError, signup, login, logout, adminLogin, adminLogout,
    theme, setTheme,
    reviews, addReview, blogPosts,
  };
  return <StoreContext.Provider value={value}>{children}</StoreContext.Provider>;
}

/* ---------------- Checkout Flow ---------------- */

function Checkout({ onClose, onComplete }) {
  const { pricing, placeOrder, discountCode, setDiscountCode, currentUser } = useStore();
  const [step, setStep] = useState(1);
  const [form, setForm] = useState({ name: currentUser?.name || "", email: currentUser?.email || "", address: "", city: "", country: "", postal: "" });
  const [payment, setPayment] = useState("card");
  const [codeInput, setCodeInput] = useState("");
  const [discountPct, setDiscountPct] = useState(0);
  const [codeMsg, setCodeMsg] = useState("");

  const totals = pricing(discountPct);

  function applyCode() {
    if (codeInput.trim().toUpperCase() === "AW10") {
      setDiscountPct(10);
      setCodeMsg("10% discount applied.");
      setDiscountCode(codeInput);
    } else {
      setDiscountPct(0);
      setCodeMsg("Code not recognized.");
    }
  }

  function handlePlaceOrder() {
    const order = placeOrder({ customer: form, payment, discountPct });
    if (order) onComplete(order);
  }

  const canContinue = step === 1
    ? form.name && form.email && form.address && form.city && form.country
    : true;

  return (
    <div className="fixed inset-0 z-[90] bg-black/75 backdrop-blur-sm flex items-center justify-center p-4" onClick={onClose}>
      <div className="bg-[#100d0a] border border-[#2a2218] rounded-2xl w-full max-w-lg max-h-[92vh] overflow-y-auto" onClick={(e) => e.stopPropagation()}>
        <div className="flex items-center justify-between p-6 border-b border-[#2a2218]">
          <h3 className="font-serif text-xl text-[#f5f0e8]">Checkout</h3>
          <button onClick={onClose} className="text-[#cbbfa8] hover:text-[#d4af6a]"><X size={20} /></button>
        </div>

        <div className="flex px-6 pt-5 gap-2">
          {[1, 2, 3].map((s) => (
            <div key={s} className={`h-1 flex-1 rounded-full ${s <= step ? "bg-[#d4af6a]" : "bg-[#2a2218]"}`} />
          ))}
        </div>

        <div className="p-6">
          {step === 1 && (
            <div className="space-y-3">
              <p className="text-[11px] tracking-wider uppercase text-[#8b7355] mb-1">Shipping Information</p>
              {[
                ["name", "Full name"], ["email", "Email address"], ["address", "Street address"],
                ["city", "City"], ["country", "Country"], ["postal", "Postal code"],
              ].map(([key, label]) => (
                <input
                  key={key}
                  placeholder={label}
                  value={form[key]}
                  onChange={(e) => setForm((f) => ({ ...f, [key]: e.target.value }))}
                  className="w-full bg-[#0c0a08] border border-[#2a2218] rounded-lg px-4 py-3 text-sm text-[#f5f0e8] placeholder-[#5a4a38] focus:outline-none focus:border-[#d4af6a]/60"
                />
              ))}
            </div>
          )}

          {step === 2 && (
            <div className="space-y-4">
              <p className="text-[11px] tracking-wider uppercase text-[#8b7355]">Payment Method</p>
              {[
                { id: "card", label: "Credit / Debit Card", icon: CreditCard, note: "Visa, Mastercard, Amex — processed securely" },
                { id: "online", label: "Online Payment", icon: Globe, note: "PayPal, Apple Pay, Google Pay" },
                { id: "cod", label: "Cash on Delivery", icon: Banknote, note: "Pay when your order arrives" },
              ].map((opt) => (
                <button
                  key={opt.id}
                  onClick={() => setPayment(opt.id)}
                  className={`w-full flex items-center gap-3 p-4 rounded-lg border text-left transition-colors ${payment === opt.id ? "border-[#d4af6a] bg-[#1a1612]" : "border-[#2a2218]"}`}
                >
                  <opt.icon size={18} className={payment === opt.id ? "text-[#d4af6a]" : "text-[#8b7355]"} />
                  <div>
                    <p className="text-sm text-[#f5f0e8]">{opt.label}</p>
                    <p className="text-xs text-[#8b7355]">{opt.note}</p>
                  </div>
                </button>
              ))}
              {payment === "card" && (
                <div className="grid grid-cols-2 gap-3 pt-1">
                  <input placeholder="Card number" className="col-span-2 bg-[#0c0a08] border border-[#2a2218] rounded-lg px-4 py-3 text-sm text-[#f5f0e8] placeholder-[#5a4a38] focus:outline-none focus:border-[#d4af6a]/60" />
                  <input placeholder="MM/YY" className="bg-[#0c0a08] border border-[#2a2218] rounded-lg px-4 py-3 text-sm text-[#f5f0e8] placeholder-[#5a4a38] focus:outline-none focus:border-[#d4af6a]/60" />
                  <input placeholder="CVC" className="bg-[#0c0a08] border border-[#2a2218] rounded-lg px-4 py-3 text-sm text-[#f5f0e8] placeholder-[#5a4a38] focus:outline-none focus:border-[#d4af6a]/60" />
                  <p className="col-span-2 text-[10px] text-[#5a4a38]">Demo only — no real payment is processed. Connect a provider like Stripe in production.</p>
                </div>
              )}
            </div>
          )}

          {step === 3 && (
            <div className="space-y-4">
              <p className="text-[11px] tracking-wider uppercase text-[#8b7355]">Order Summary</p>
              <div className="flex gap-2">
                <input
                  placeholder="Discount code (try AW10)"
                  value={codeInput}
                  onChange={(e) => setCodeInput(e.target.value)}
                  className="flex-1 bg-[#0c0a08] border border-[#2a2218] rounded-lg px-4 py-2.5 text-sm text-[#f5f0e8] placeholder-[#5a4a38] focus:outline-none focus:border-[#d4af6a]/60"
                />
                <button onClick={applyCode} className="px-4 text-xs uppercase tracking-wider border border-[#d4af6a]/50 text-[#d4af6a] rounded-lg hover:bg-[#d4af6a] hover:text-[#0a0908] transition-colors">Apply</button>
              </div>
              {codeMsg && <p className="text-xs text-[#8b7355]">{codeMsg}</p>}
              <div className="space-y-2 pt-2 border-t border-[#2a2218]">
                <div className="flex justify-between text-sm text-[#cbbfa8]"><span>Subtotal</span><span><Price value={totals.subtotal} /></span></div>
                {totals.discount > 0 && <div className="flex justify-between text-sm text-[#d4af6a]"><span>Discount</span><span>-<Price value={totals.discount} /></span></div>}
                <div className="flex justify-between text-sm text-[#cbbfa8]"><span>Shipping</span><span>{totals.shipping === 0 ? "Free" : <Price value={totals.shipping} />}</span></div>
                <div className="flex justify-between text-sm text-[#cbbfa8]"><span>Tax</span><span><Price value={totals.tax} /></span></div>
                <div className="flex justify-between font-serif text-lg text-[#f5f0e8] pt-2 border-t border-[#2a2218]">
                  <span>Total</span><span className="text-[#d4af6a]"><Price value={totals.total} /></span>
                </div>
              </div>
              <p className="text-xs text-[#8b7355]">Shipping to {form.address}, {form.city}, {form.country}. Paying via {payment === "card" ? "Card" : payment === "cod" ? "Cash on Delivery" : "Online Payment"}.</p>
            </div>
          )}
        </div>

        <div className="p-6 pt-0 flex gap-3">
          {step > 1 && (
            <button onClick={() => setStep((s) => s - 1)} className="flex-1 border border-[#2a2218] text-[#cbbfa8] text-xs uppercase tracking-wider py-3 rounded-full">Back</button>
          )}
          {step < 3 ? (
            <button
              disabled={!canContinue}
              onClick={() => setStep((s) => s + 1)}
              className="flex-1 bg-[#d4af6a] disabled:opacity-40 text-[#0a0908] text-xs uppercase tracking-wider py-3 rounded-full font-medium"
            >
              Continue
            </button>
          ) : (
            <button onClick={handlePlaceOrder} className="flex-1 bg-[#d4af6a] text-[#0a0908] text-xs uppercase tracking-wider py-3 rounded-full font-medium">
              Place Order
            </button>
          )}
        </div>
      </div>
    </div>
  );
}

/* ---------------- Customer Auth Modal (Login / Signup) ---------------- */

function AuthModal({ open, onClose, mode, setMode }) {
  const { signup, login, authError } = useStore();
  const [form, setForm] = useState({ name: "", email: "", password: "" });

  if (!open) return null;

  function handleSubmit(e) {
    e.preventDefault();
    const ok = mode === "signup" ? signup(form) : login(form);
    if (ok) { onClose(); setForm({ name: "", email: "", password: "" }); }
  }

  return (
    <div className="fixed inset-0 z-[95] bg-black/75 backdrop-blur-sm flex items-center justify-center p-4" onClick={onClose}>
      <div className="bg-[#100d0a] border border-[#2a2218] rounded-2xl w-full max-w-sm p-7" onClick={(e) => e.stopPropagation()}>
        <div className="flex items-center justify-between mb-1">
          <div className="flex items-center gap-2">
            <Lock size={15} className="text-[#d4af6a]" />
            <Eyebrow>Account Security</Eyebrow>
          </div>
          <button onClick={onClose} className="text-[#cbbfa8] hover:text-[#d4af6a]"><X size={18} /></button>
        </div>
        <h3 className="font-serif text-2xl text-[#f5f0e8] mb-5">{mode === "signup" ? "Create Your Account" : "Welcome Back"}</h3>

        <form onSubmit={handleSubmit} className="space-y-3">
          {mode === "signup" && (
            <input
              placeholder="Full name"
              value={form.name}
              onChange={(e) => setForm((f) => ({ ...f, name: e.target.value }))}
              className="w-full bg-[#0c0a08] border border-[#2a2218] rounded-lg px-4 py-3 text-sm text-[#f5f0e8] placeholder-[#5a4a38] focus:outline-none focus:border-[#d4af6a]/60"
            />
          )}
          <input
            type="email"
            placeholder="Email address"
            value={form.email}
            onChange={(e) => setForm((f) => ({ ...f, email: e.target.value }))}
            className="w-full bg-[#0c0a08] border border-[#2a2218] rounded-lg px-4 py-3 text-sm text-[#f5f0e8] placeholder-[#5a4a38] focus:outline-none focus:border-[#d4af6a]/60"
          />
          <input
            type="password"
            placeholder="Password"
            value={form.password}
            onChange={(e) => setForm((f) => ({ ...f, password: e.target.value }))}
            className="w-full bg-[#0c0a08] border border-[#2a2218] rounded-lg px-4 py-3 text-sm text-[#f5f0e8] placeholder-[#5a4a38] focus:outline-none focus:border-[#d4af6a]/60"
          />
          {authError && <p className="text-xs text-[#e07a5f] flex items-center gap-1"><AlertCircle size={12} /> {authError}</p>}
          <button type="submit" className="w-full bg-[#d4af6a] text-[#0a0908] text-xs uppercase tracking-wider py-3 rounded-full font-medium mt-2">
            {mode === "signup" ? "Create Account" : "Log In"}

/* ---------------- AI Assistant Chatbot ---------------- */

function answerAssistant(message, catalog) {
  const m = message.toLowerCase();
  if (/(track|order status|where is my order|delivery)/.test(m)) {
    return "I can help with that — once your order is placed, you'll get an order ID like AW-1000. You can check its live status under Account → Orders, or just tell me the order ID here.";
  }
  if (/(return|refund|exchange)/.test(m)) {
    return "We offer free returns within 30 days on unopened bottles. I can start that for you — just share your order ID and I'll flag it for our team.";
  }
  if (/(gift|present)/.test(m)) {
    return "For a gift, I'd lean toward something warm and memorable — Ambre Royale or Velours d'Or are our most-gifted scents. Want me to suggest one based on who it's for?";
  }
  if (/(shipping|deliver)/.test(m)) {
    return "We ship worldwide. Orders over $250 ship free, otherwise it's a flat $18, arriving in 3–7 business days depending on destination.";
  }
  if (/(ingredient|allergen|vegan|cruelty)/.test(m)) {
    return "All A_W fragrances are cruelty-free and we publish full ingredient lists on each product page under 'Fragrance Notes'. Let me know a specific bottle and I'll pull the details.";
  }
  const found = catalog.find((p) => m.includes(p.name.toLowerCase()));
  if (found) {
    return `${found.name} is ${found.profile.mood.toLowerCase()} and built around ${found.profile.top[0]} up top with a ${found.profile.base[0]} base — best suited for ${found.profile.occasion.toLowerCase()}. It's $${found.price} for ${found.sizeMl}ml. Want me to add it to your bag?`;
  }
  return "I'm here to help with sizing, notes, shipping, or finding your next signature scent. Try the 'Suggest Me a Perfume' tool below, or ask me about a specific fragrance.";
}

/* ---------------- AI Perfume Suggestor (signature element) ---------------- */

function ScentProfiler({ onResult }) {
  const { catalog } = useStore();
  const [mood, setMood] = useState(null);
  const [gender, setGender] = useState(null);
  const [occasion, setOccasion] = useState(null);
  const [result, setResult] = useState(null);

  function compute() {
    let best = catalog[0];
    let bestScore = -1;
    catalog.forEach((p) => {
      let score = 0;
      if (mood && p.profile.mood === mood) score += 3;
      if (gender && (p.profile.gender === gender || p.profile.gender === "Unisex")) score += 2;
      if (occasion && p.profile.occasion === occasion) score += 3;
      if (score > bestScore) { bestScore = score; best = p; }
    });
    setResult(best);
    onResult?.(best);
  }

  return (
    <div className="bg-gradient-to-br from-[#15120e] to-[#0c0a08] border border-[#2a2218] rounded-2xl p-6 md:p-8">
      <div className="flex items-center gap-2 mb-1">
        <Wand2 size={16} className="text-[#d4af6a]" />
        <Eyebrow>AI Scent Profiler</Eyebrow>
      </div>
      <h3 className="font-serif text-2xl text-[#f5f0e8] mb-5">Find Your Signature Scent</h3>

      <div className="space-y-4">
        <ChipRow label="Mood" options={MOODS} value={mood} onChange={setMood} />
        <ChipRow label="For" options={GENDERS} value={gender} onChange={setGender} />
        <ChipRow label="Occasion" options={OCCASIONS} value={occasion} onChange={setOccasion} />
      </div>

      <button
        onClick={compute}
        disabled={!mood && !gender && !occasion}
        className="mt-6 w-full sm:w-auto px-7 bg-[#d4af6a] disabled:opacity-40 text-[#0a0908] text-xs uppercase tracking-wider py-3 rounded-full font-medium"
      >
        Reveal My Match
      </button>

      {result && (
        <div className="mt-7 pt-6 border-t border-[#2a2218] flex flex-col sm:flex-row gap-5 items-center sm:items-start">
          <div className="w-20 h-28 flex-shrink-0 flex items-center justify-center bg-[#0c0a08] rounded-lg border border-[#2a2218]">
            <BottleArt art={result.art} size="small" />
          </div>
          <div className="flex-1">
            <p className="text-[10px] uppercase tracking-wider text-[#8b7355]">Your match</p>
            <p className="font-serif text-xl text-[#f5f0e8]">{result.name}</p>
            <p className="text-sm text-[#cbbfa8] mt-1 leading-relaxed">{result.description}</p>
            <ScentDNA profile={result.profile} />
          </div>
        </div>
      )}
    </div>
  );
}

/* ---------------- Smart Recommendations ---------------- */

function SmartRecommendations({ onOpen }) {
  const { catalog, viewHistory, wishlist } = useStore();
  const recs = useMemo(() => {
    const seedIds = [...viewHistory, ...wishlist];
    if (seedIds.length === 0) return catalog.filter((p) => p.featured).slice(0, 4);
    const moods = new Set(seedIds.map((id) => catalog.find((p) => p.id === id)?.profile.mood).filter(Boolean));
    const ranked = catalog
      .filter((p) => !seedIds.includes(p.id))
      .map((p) => ({ p, score: moods.has(p.profile.mood) ? 1 : 0 }))
      .sort((a, b) => b.score - a.score)
      .map((x) => x.p);
    return ranked.slice(0, 4);
  }, [catalog, viewHistory, wishlist]);

  return (
    <section className="px-6 md:px-12 py-16 border-t border-[#1a1612]">
      <div className="flex items-center gap-2 mb-1">
        <TrendingUp size={15} className="text-[#d4af6a]" />
        <Eyebrow>{viewHistory.length ? "Curated For You" : "Editor's Picks"}</Eyebrow>
      </div>
      <h2 className="font-serif text-3xl text-[#f5f0e8] mb-8">{viewHistory.length ? "Because You Browsed These Notes" : "Where to Begin"}</h2>
      <div className="grid grid-cols-2 md:grid-cols-4 gap-6 md:gap-8">
        {recs.map((p) => <ProductCard key={p.id} product={p} onOpen={onOpen} />)}
      </div>
    </section>
  );
}

function ShopPage({ onOpen }) {
  const { catalog } = useStore();
  const [query, setQuery] = useState("");
  const [genderFilter, setGenderFilter] = useState("All");
  const [sort, setSort] = useState("featured");

  let items = catalog.filter((p) =>
    p.name.toLowerCase().includes(query.toLowerCase()) &&
    (genderFilter === "All" || p.profile.gender === genderFilter)
  );
  if (sort === "price-asc") items = [...items].sort((a, b) => a.price - b.price);
  if (sort === "price-desc") items = [...items].sort((a, b) => b.price - a.price);
  if (sort === "rating") items = [...items].sort((a, b) => b.rating - a.rating);

  return (
    <section className="px-6 md:px-12 py-12 max-w-7xl mx-auto">
      <Eyebrow>Full Collection</Eyebrow>
      <h1 className="font-serif text-4xl text-[#f5f0e8] mb-8">Every Composition</h1>

      <div className="flex flex-wrap gap-3 mb-10 items-center">
        <div className="relative flex-1 min-w-[200px]">
          <Search size={15} className="absolute left-3.5 top-1/2 -translate-y-1/2 text-[#5a4a38]" />
          <input
            value={query}
            onChange={(e) => setQuery(e.target.value)}
            placeholder="Search fragrances..."
            className="w-full bg-[#100d0a] border border-[#2a2218] rounded-full pl-10 pr-4 py-2.5 text-sm text-[#f5f0e8] placeholder-[#5a4a38] focus:outline-none focus:border-[#d4af6a]/60"
          />
        </div>
        <select value={genderFilter} onChange={(e) => setGenderFilter(e.target.value)} className="bg-[#100d0a] border border-[#2a2218] rounded-full px-4 py-2.5 text-sm text-[#cbbfa8] focus:outline-none">
          {["All", ...GENDERS].map((g) => <option key={g}>{g}</option>)}
        </select>
        <select value={sort} onChange={(e) => setSort(e.target.value)} className="bg-[#100d0a] border border-[#2a2218] rounded-full px-4 py-2.5 text-sm text-[#cbbfa8] focus:outline-none">
          <option value="featured">Featured</option>
          <option value="price-asc">Price: Low to High</option>
          <option value="price-desc">Price: High to Low</option>
          <option value="rating">Top Rated</option>
        </select>
      </div>

      <div className="grid grid-cols-2 md:grid-cols-4 gap-6 md:gap-8">
        {items.map((p) => <ProductCard key={p.id} product={p} onOpen={onOpen} />)}
      </div>
      {items.length === 0 && <p className="text-[#8b7355] text-sm text-center py-20">No fragrances match your search.</p>}
    </section>
  );
}

function OrderRow({ order, onAdvance, onInvoice, expanded }) {
  const [open, setOpen] = useState(false);
  const currentIdx = STATUS_FLOW.indexOf(order.status);
  const Icon = STATUS_ICON[order.status];
  return (
    <div className="border border-[#1a1612] rounded-xl bg-[#0c0a08]">
      <button onClick={() => setOpen((o) => !o)} className="w-full flex items-center justify-between p-4 text-left">
        <div className="flex items-center gap-3">
          <Icon size={16} className="text-[#d4af6a]" />
          <div>
            <p className="text-sm text-[#f5f0e8]">{order.id} <span className="text-[#8b7355]">· {order.customer.name || "Guest"}</span></p>
            <p className="text-[10px] text-[#8b7355]">{new Date(order.date).toLocaleString()}</p>
          </div>
        </div>
        <div className="flex items-center gap-3">
          <span className="text-sm text-[#d4af6a]">${order.totals.total.toFixed(2)}</span>
          <span className="text-[10px] uppercase tracking-wider px-2.5 py-1 rounded-full bg-[#1a1612] text-[#cbbfa8]">{order.status}</span>
        </div>
      </button>
      {(open || expanded) && (
        <div className="px-4 pb-4 space-y-3">
          <div className="text-xs text-[#8b7355]">
            {order.items.map((it) => (
              <div key={it.id} className="flex justify-between py-1">
                <span>{it.name} × {it.qty}</span><span>${(it.price * it.qty).toFixed(2)}</span>
              </div>
            ))}
          </div>
          <div className="flex items-center gap-2 flex-wrap">
            {STATUS_FLOW.map((s, i) => (
              <button
                key={s}
                onClick={() => onAdvance(order.id, s)}
                disabled={i < currentIdx}
                className={`text-[10px] uppercase tracking-wider px-3 py-1.5 rounded-full border ${i === currentIdx ? "border-[#d4af6a] bg-[#d4af6a] text-[#0a0908]" : i < currentIdx ? "border-[#2a2218] text-[#5a4a38]" : "border-[#2a2218] text-[#cbbfa8]"}`}
              >
                {s}
              </button>
            ))}
            <button onClick={() => onInvoice(order)} className="text-[10px] uppercase tracking-wider px-3 py-1.5 rounded-full border border-[#2a2218] text-[#d4af6a] flex items-center gap-1 ml-auto">
              <FileText size={11} /> Invoice
            </button>
          </div>
        </div>
      )}
    </div>
  );
}

/* ---------------- Product Reviews Component ---------------- */

function ReviewsSection({ productId }) {
  const { reviews, addReview, currentUser } = useStore();
  const productReviews = reviews[productId] || [];
  const [form, setForm] = useState({ rating: 5, comment: "" });
  const [submitted, setSubmitted] = useState(false);

  function handleSubmit() {
    if (!form.comment.trim()) return;
    addReview(productId, {
      name: currentUser?.name || "Anonymous",
      rating: form.rating,
      comment: form.comment,
    });
    setForm({ rating: 5, comment: "" });
    setSubmitted(true);
    setTimeout(() => setSubmitted(false), 2000);
  }

  return (
    <div className="mt-6 pt-5 border-t border-[#2a2218]">
      <p className="text-[10px] uppercase tracking-wider text-[#8b7355] mb-3">Customer Reviews ({productReviews.length})</p>

      {/* Write a review */}
      <div className="bg-[#0c0a08] border border-[#2a2218] rounded-lg p-4 mb-4">
        <p className="text-xs text-[#cbbfa8] mb-2">Write a Review</p>
        <div className="flex gap-1 mb-2">
          {[1,2,3,4,5].map((s) => (
            <button key={s} onClick={() => setForm((f) => ({ ...f, rating: s }))}>
              <Star size={16} className={s <= form.rating ? "fill-[#d4af6a] text-[#d4af6a]" : "text-[#2a2218]"} />
            </button>
          ))}
        </div>
        <textarea
          value={form.comment}
          onChange={(e) => setForm((f) => ({ ...f, comment: e.target.value }))}
          placeholder="Share your experience..."
          rows={2}
          className="w-full bg-transparent border border-[#2a2218] rounded-lg px-3 py-2 text-sm text-[#f5f0e8] placeholder-[#5a4a38] focus:outline-none focus:border-[#d4af6a]/60 resize-none"
        />
        <button
          onClick={handleSubmit}
          className="mt-2 text-[10px] uppercase tracking-wider bg-[#d4af6a] text-[#0a0908] px-4 py-1.5 rounded-full font-medium"
        >
          {submitted ? "Posted ✓" : "Post Review"}
        </button>
      </div>

      {/* Existing reviews */}
      <div className="space-y-3 max-h-48 overflow-y-auto pr-1">
        {productReviews.length === 0 ? (
          <p className="text-xs text-[#5a4a38]">No reviews yet — be the first!</p>
        ) : productReviews.map((r, i) => (
          <div key={i} className="border-b border-[#1a1612] pb-3">
            <div className="flex items-center justify-between mb-1">
              <div className="flex items-center gap-2">
                <div className="w-6 h-6 rounded-full bg-[#1a1612] flex items-center justify-center text-[10px] text-[#d4af6a]">{r.name[0]}</div>
                <span className="text-xs text-[#cbbfa8]">{r.name}</span>

              <div className="flex gap-0.5">
                {[1,2,3,4,5].map((s) => <Star key={s} size={10} className={s <= r.rating ? "fill-[#d4af6a] text-[#d4af6a]" : "text-[#2a2218]"} />)}
              </div>
            </div>
            <p className="text-xs text-[#8b7355] leading-relaxed">{r.comment}</p>
            <p className="text-[10px] text-[#5a4a38] mt-1">{new Date(r.date).toLocaleDateString()}</p>

/* ---------------- Root App ---------------- */

function AppInner() {
  const [page, setPage] = useState("home");
  const [cartOpen, setCartOpen] = useState(false);
  const [checkoutOpen, setCheckoutOpen] = useState(false);
  const [activeProduct, setActiveProduct] = useState(null);
  const [confirmedOrder, setConfirmedOrder] = useState(null);
  const [aiOpen, setAiOpen] = useState(false);
  const [authOpen, setAuthOpen] = useState(false);
  const [authMode, setAuthMode] = useState("login");
  const { cart, currentUser, isAdmin, theme } = useStore();

  const cartCount = cart.reduce((s, i) => s + i.qty, 0);

  function handleNav(p) {
    setPage(p);
    window.scrollTo({ top: 0, behavior: "smooth" });
  }

  function handleCheckoutRequest() {
    setCartOpen(false);
    if (!currentUser) {
      setAuthMode("login");
      setAuthOpen(true);
      return;
    }
    setCheckoutOpen(true);
  }

  return (
    <div className="min-h-screen bg-[#0a0908] text-[#e8e1d3]" style={{ fontFamily: "'Inter', sans-serif", "--accent": theme.accent }}>
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Cormorant+Garamond:ital,wght@0,400;0,500;0,600;1,400&family=Inter:wght@300;400;500;600&display=swap');
        .font-serif { font-family: 'Cormorant Garamond', serif; }
        ::selection { background: ${theme.accent}; color: #0a0908; }
        * { scrollbar-width: thin; scrollbar-color: #2a2218 #0a0908; }
      `}</style>

      <Header cartCount={cartCount} onCartOpen={() => setCartOpen(true)} onNav={handleNav} page={page} onAuthOpen={() => { setAuthMode("login"); setAuthOpen(true); }} />

      {page === "home" && <HomePage onOpen={setActiveProduct} onNav={handleNav} />}
      {page === "shop" && <ShopPage onOpen={setActiveProduct} />}
      {page === "profiler" && <ProfilerPage onOpen={setActiveProduct} />}
      {page === "blog" && <BlogPage />}
      {page === "account" && <AccountPage onAuthOpen={() => { setAuthMode("login"); setAuthOpen(true); }} />}
      {page === "admin" && (isAdmin ? <AdminDashboard /> : <AdminGate onSuccess={() => {}} onCancel={() => handleNav("home")} />)}

      {page !== "admin" && <Footer />}

      <ProductModal product={activeProduct} onClose={() => setActiveProduct(null)} />
      <CartDrawer open={cartOpen} onClose={() => setCartOpen(false)} onCheckout={handleCheckoutRequest} />
      {checkoutOpen && (
        <Checkout
          onClose={() => setCheckoutOpen(false)}
          onComplete={(order) => { setCheckoutOpen(false); setConfirmedOrder(order); }}
        />
      )}
      <OrderConfirmation
        order={confirmedOrder}
        onClose={() => setConfirmedOrder(null)}
        onTrack={() => { setConfirmedOrder(null); handleNav("admin"); }}
      />
      <AuthModal open={authOpen} onClose={() => setAuthOpen(false)} mode={authMode} setMode={setAuthMode} />
      <AIAssistant open={aiOpen} onToggle={() => setAiOpen((o) => !o)} />
    </div>
  );
}

export default function App() {
  return (
    <StoreProvider>
      <AppInner />
    </StoreProvider>
  );
}

/* ============================================================
README — WHAT'S REAL VS SIMULATED IN THIS DEMO

This build is fully interactive: cart, checkout, order placement,
invoice generation, admin dashboard with live order status, stock
management, and the AI scent profiler all genuinely work using
in-memory state (StoreProvider above).

To take this to a real production business, replace these pieces:

1. AI PRODUCT IMAGES: BottleArt is a CSS-generated placeholder.
   Swap in calls to an image model (e.g. via the Anthropic API
   using a prompt built from each product's fragrance profile) and
   store the resulting URLs against each product.

2. AI DESCRIPTIONS/SEO: ✅ LIVE. generateDescriptionAndSEO_AI() calls the
   real Anthropic API (Claude) to write unique descriptions and SEO
   metadata for any product, with a local template fallback if the
   call fails. Used by "Regenerate" on existing products and by the
   "AI Product Builder" panel for creating brand-new fragrances from
   just a name.

3. CHATBOT: answerAssistant is rule-based. Replace with a real LLM
   call carrying conversation history + a system prompt describing
   A_W's catalog, policies, and tone.

4. PERSISTENCE: All state (catalog, orders, cart) lives in React
   state and resets on reload. Replace with a real database
   (Postgres/Supabase/Firebase) and an authenticated backend API.

5. PAYMENTS: The checkout UI collects card fields but does not
   process them. Integrate Stripe/PayPal/Razorpay server-side —
   never handle raw card numbers client-side in production.

6. AUTH: ⚠️ DEMO-LEVEL ONLY. Customer signup/login and the admin
   login gate are now implemented (StoreProvider: signup, login,
   adminLogin) — but passwords are only run through simpleHash(),
   a basic string transform, NOT real cryptographic hashing, and
   everything lives in browser memory (resets on reload, visible
   in React state/devtools). This is NOT secure against real
   attackers. Before going live:
     - Move auth to a real backend (Node/Express, Supabase Auth,
       Firebase Auth, Clerk, or Auth0).
     - Hash passwords server-side with bcrypt or argon2 — never
       compare passwords in client-side JS.
     - Store sessions as HTTP-only cookies or signed JWTs, not in
       React state.
     - Rate-limit login attempts and add 2FA for the admin account.
     - Change the demo admin password (admin@awperfume.com /
       AWAdmin#2026) immediately — it is hardcoded in this file and
       visible to anyone who reads the source.

8. CUSTOMIZATION: The "Customize" tab in Admin (CustomizePanel) lets
   you change the store name, tagline, and accent color live. This
   is also session-only (React state) — to persist branding across
   reloads/users, save the theme object to your backend/database and
   load it on app start instead of the local default in StoreProvider.

7. EMAIL: Order confirmation and invoice are shown in-app only.
   Wire up transactional email (e.g. via SendGrid/Postmark) to
   actually email customers their invoice PDF.
============================================================ */
